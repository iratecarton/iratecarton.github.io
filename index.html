#include

#define trigPin 5

#define echoPin 4


float duration;

float distance;


int pinServo1;

int pinServo2;

int pinServo3;

int pinServo4;


int servoSpeed1;

int servoSpeed2;

int servoSpeed3;

int servoSpeed4;


int servoStop1;

int servoStop2;

int servoStop3;

int servoStop4;


int speed1;

int speed2;

int speed3;

int speed4;


Servo FRServo;

Servo FLServo;

Servo BRServo;

Servo BLServo;

void setup() {


Serial.begin (9600);


pinMode (trigPin, OUTPUT);

pinMode (echoPin, INPUT);


speed1 = -1000;

speed2 = 1000;

speed3 = -1000;

speed4 = 1000;


pinServo1 = 13;

pinServo2 = 10;

pinServo3 = 11;

pinServo4 = 12;


FRServo.attach(pinServo1);

FLServo.attach(pinServo2);

BRServo.attach(pinServo3);

BLServo.attach(pinServo4);


servoStop1 = 1500;

servoStop2 = 1500;

servoStop3 = 1500;

servoStop4 = 1500;


}

void loop() {


digitalWrite(trigPin, LOW);

delayMicroseconds(2);

digitalWrite(trigPin, HIGH);

delayMicroseconds(10);

digitalWrite(trigPin, LOW);


duration = pulseIn(echoPin, HIGH);

distance = (duration / 2) * 0.0343;


servoSpeed1 = servoStop1 + speed1;

servoSpeed2 = servoStop2 + speed2;

servoSpeed3 = servoStop3 + speed3;

servoSpeed4 = servoStop4 + speed4;


runServo1(servoSpeed1);

runServo2(servoSpeed2);

runServo3(servoSpeed3);

runServo4(servoSpeed4);


if (distance <= 15){

speed1 = 0;

speed2 = 0;

speed3 = 0;

speed4 = 0;
}
else {

speed1 = -1000;

speed2 = 1000;

speed3 = -1000;

speed4 = 1000;
}
}

void runServo1(int spd){

FRServo.writeMicroseconds(spd);

}

void runServo2(int spd){

FLServo.writeMicroseconds(spd);

}

void runServo3(int spd){

BRServo.writeMicroseconds(spd);

}

void runServo4(int spd){

BLServo.writeMicroseconds(spd);

}
